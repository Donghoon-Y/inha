def runge_kutta4(f, tspan, y0, h):
    
    t = np.arange(tspan[0], tspan[1], h)
    t = np.insert(t, len(t), tspan[1])

    y = np.empty((len(t), len(y0))) 
    y[0] = y0

    for i, ti in enumerate(t):
        if i < len(t) - 1:           
            k1 = f(ti,           y[i])
            k2 = f(ti + 1/2*h,   y[i] + 1/2*k1*h)
            k3 = f(ti + 1/2*h,   y[i] + 1/2*k2*h)
            k4 = f(ti + h,       y[i] + k3*h)
            y[i+1] = y[i] + h*(k1 + 2*k2 + 2*k3 + k4)/6

    return t, y

Assumption and Initial Condition¶

Assumption

  • Constant speed(No drag)
  • Acceleration normal to velocity

Initial Condition

  • $V = V_0 = 200m/s$
  • $x_0 = y_0 = 0 $
  • ${\psi_0} = 0$
  • $a = 50 m/s^2$

Rectangular formulation¶

$ \dot{x} = V_x , x(0) = x_0 \\$ $ \dot{y} = V_y , y(0) = y_0 \ $ $\dot{V_x} = acos{\psi}, V_x(0) = Vsin{\psi_0}\$ $\dot{V_y} = -asin{\psi}, V_y(0) = Vcos{\psi_0}\\$ $\text{where}: \psi = tan^{-1}\frac{V_x}{V_y}$

def make_f(u):
    def f(t, x):
        [Px, Py, Vx, Vy] = x
        a = u

        psi = np.arctan2(Vx, Vy)
        if psi < 0:
            psi += 2 * np.pi

        Vx_dot = a * np.cos(psi)
        Vy_dot = -a * np.sin(psi)

        return np.array([Vx, Vy, Vx_dot, Vy_dot])
    return f

Polar formulation¶

$ \dot{x} = V_x , x(0) = x_0 \\$ $ \dot{y} = V_y , y(0) = y_0 \ $ $\dot{V} = 0, V(0) = V\$ $\dot{\psi} = \frac{a}{V}, \psi(0) = \psi_0$

def make_f(u):
    def f(t, x):
        [Px, Py, V, Psi] = x
        a = u

        Vx = V*np.sin(Psi)
        Vy = V*np.cos(Psi)
        V_dot = 0
        Psi_dot = a/V

        return np.array([Vx, Vy, V_dot, Psi_dot])
    return f
